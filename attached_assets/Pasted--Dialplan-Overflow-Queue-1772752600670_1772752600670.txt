<!--
  ================================================================
  Dialplan: Overflow Queue - Alessia AI Voice Platform
  ================================================================
  
  Gestisce le chiamate inbound quando tutte le linee AI sono occupate.
  Il Voice Bridge trasferisce la chiamata qui via uuid_transfer quando
  il limite max_concurrent viene raggiunto.
  
  VARIABILI CHANNEL (impostate dal Voice Bridge via uuid_setvar):
  - fallback_transfer_number: numero di trasferimento del consulente
  - overflow_dtmf_enabled: true/false - abilita "premi 1"
  - sip_gateway: gateway SIP da usare per il trasferimento
  - tech_prefix: prefisso tecnico per il gateway (es. 0312 per Telnyx)
  - queue_position: posizione nella coda (1, 2, 3...)
  - overflow_custom_message: messaggio personalizzato (opzionale)
  
  FLUSSO:
  1. Answer difensivo (se canale non già answered)
  2. Bind DTMF "1" per trasferimento (se overflow_dtmf_enabled=true)
  3. Ciclo 1: posizione + hold_A (social proof)
  4. Ciclo 2: posizione + hold_B (urgenza morbida)
  5. Ciclo 3: posizione + hold_C (curiosità / hook)
  6. Ciclo 4: posizione + hold_D (valore gratuito)
  7. Ciclo 5: posizione + hold_E (identità brand)
  8. Ciclo 6: posizione + hold_A (riparte la rotazione)
  9. Timeout → park (il Voice Bridge gestisce il timeout)
  
  FILE AUDIO in /opt/sounds/overflow/:
  OBBLIGATORI per esperienza ottimale:
  - hold_A.wav: messaggio social proof (~30-40 sec)
  - hold_B.wav: messaggio urgenza morbida (~30-40 sec)
  - hold_C.wav: messaggio curiosità / hook (~30-40 sec)
  - hold_D.wav: messaggio valore gratuito (~30-40 sec)
  - hold_E.wav: messaggio identità brand (~30-40 sec)
  
  OPZIONALI (fallback automatico se assenti):
  - announcement.wav: annuncio principale (usato solo se i hold_X mancano)
  - transferring.wav: "Ti stiamo trasferendo..."
  - transfer_failed.wav: "Il trasferimento non è riuscito"
  - timeout.wav: messaggio pre-disconnessione per timeout
  - position_prefix.wav: "Sei in posizione numero"
  - position_suffix.wav: "nella coda d'attesa"
  
  PERCORSO DEPLOY:
  /usr/local/freeswitch/etc/freeswitch/dialplan/default/05_overflow_queue.xml
  ================================================================
-->

<include>

  <extension name="overflow_queue">
    <condition field="destination_number" expression="^overflow_queue$">

      <action application="log" data="INFO [OVERFLOW] Chiamata ${uuid} entrata in coda overflow (fallback=${fallback_transfer_number}, dtmf=${overflow_dtmf_enabled}, pos=${queue_position})"/>

      <!-- Answer difensivo: se il canale non è già answered, lo risponde -->
      <action application="answer"/>

      <action application="set" data="playback_terminators=none"/>

      <!-- Bind DTMF "1" per trasferimento manuale -->
      <action application="bind_digit_action" data="overflow,1,exec:transfer,transfer_to_fallback XML default"/>
      <action application="digit_action_set_realm" data="overflow"/>

      <!-- Ciclo 1: posizione + social proof -->
      <action application="execute_extension" data="announce_position XML default" inline="true"/>
      <action application="execute_extension" data="play_hold_A XML default" inline="true"/>

      <!-- Ciclo 2: posizione + urgenza morbida -->
      <action application="execute_extension" data="announce_position XML default" inline="true"/>
      <action application="execute_extension" data="play_hold_B XML default" inline="true"/>

      <!-- Ciclo 3: posizione + curiosità hook -->
      <action application="execute_extension" data="announce_position XML default" inline="true"/>
      <action application="execute_extension" data="play_hold_C XML default" inline="true"/>

      <!-- Ciclo 4: posizione + valore gratuito -->
      <action application="execute_extension" data="announce_position XML default" inline="true"/>
      <action application="execute_extension" data="play_hold_D XML default" inline="true"/>

      <!-- Ciclo 5: posizione + identità brand -->
      <action application="execute_extension" data="announce_position XML default" inline="true"/>
      <action application="execute_extension" data="play_hold_E XML default" inline="true"/>

      <!-- Ciclo 6: riparte la rotazione con A -->
      <action application="execute_extension" data="announce_position XML default" inline="true"/>
      <action application="execute_extension" data="play_hold_A XML default" inline="true"/>

      <action application="log" data="WARNING [OVERFLOW] Chiamata ${uuid} - loop completato senza dequeue, parking"/>
      <action application="park"/>

    </condition>
  </extension>

  <!-- ============================================================
       Sub-extension: annuncia posizione in coda
       Riproduce position_prefix.wav + numero TTS + position_suffix.wav
       Se i WAV non esistono, salta silenziosamente
  ============================================================ -->
  <extension name="announce_position">
    <condition field="destination_number" expression="^announce_position$" break="never">
      <action application="log" data="INFO [OVERFLOW] announce_position chiamato, queue_position=${queue_position}"/>
    </condition>
    <condition field="destination_number" expression="^announce_position$" field2="${queue_position}" expression2="^[1-9]\d*$">
      <action application="playback" data="/opt/sounds/overflow/position_prefix.wav"/>
      <action application="say" data="it NUMBER PRONOUNCED ${queue_position}"/>
      <action application="playback" data="/opt/sounds/overflow/position_suffix.wav"/>
      <anti-action application="log" data="INFO [OVERFLOW] Skipping position: queue_position non valida o WAV assenti"/>
    </condition>
  </extension>

  <!-- ============================================================
       Sub-extension: hold_A — Social proof
       "Ogni giorno persone come te scelgono {{businessName}}..."
  ============================================================ -->
  <extension name="play_hold_A">
    <condition field="destination_number" expression="^play_hold_A$">
      <action application="playback" data="/opt/sounds/overflow/hold_A.wav"/>
      <anti-action application="playback" data="local_stream://moh"/>
    </condition>
  </extension>

  <!-- ============================================================
       Sub-extension: hold_B — Urgenza morbida
       "Sappiamo che il tuo tempo è prezioso..."
  ============================================================ -->
  <extension name="play_hold_B">
    <condition field="destination_number" expression="^play_hold_B$">
      <action application="playback" data="/opt/sounds/overflow/hold_B.wav"/>
      <anti-action application="playback" data="local_stream://moh"/>
    </condition>
  </extension>

  <!-- ============================================================
       Sub-extension: hold_C — Curiosità / hook
       "Ti stai chiedendo se {{businessName}} fa al caso tuo?..."
  ============================================================ -->
  <extension name="play_hold_C">
    <condition field="destination_number" expression="^play_hold_C$">
      <action application="playback" data="/opt/sounds/overflow/hold_C.wav"/>
      <anti-action application="playback" data="local_stream://moh"/>
    </condition>
  </extension>

  <!-- ============================================================
       Sub-extension: hold_D — Valore gratuito
       "La chiamata che stai per fare è completamente gratuita..."
  ============================================================ -->
  <extension name="play_hold_D">
    <condition field="destination_number" expression="^play_hold_D$">
      <action application="playback" data="/opt/sounds/overflow/hold_D.wav"/>
      <anti-action application="playback" data="local_stream://moh"/>
    </condition>
  </extension>

  <!-- ============================================================
       Sub-extension: hold_E — Identità brand
       "{{businessName}} nasce con un'idea semplice: parlare chiaro..."
  ============================================================ -->
  <extension name="play_hold_E">
    <condition field="destination_number" expression="^play_hold_E$">
      <action application="playback" data="/opt/sounds/overflow/hold_E.wav"/>
      <anti-action application="playback" data="local_stream://moh"/>
    </condition>
  </extension>

  <!-- ============================================================
       Extension: timeout — messaggio pre-disconnessione
       Chiamato dal Voice Bridge via uuid_transfer quando scade il timeout
  ============================================================ -->
  <extension name="overflow_timeout">
    <condition field="destination_number" expression="^overflow_timeout$">
      <action application="log" data="INFO [OVERFLOW-TIMEOUT] Chiamata ${uuid} - timeout raggiunto"/>
      <action application="clear_digit_action" data="overflow"/>
      <action application="playback" data="/opt/sounds/overflow/timeout.wav"/>
      <anti-action application="playback" data="tone_stream://%(500,100,400);%(500,100,300);%(1000,0,200)"/>
      <action application="hangup" data="NORMAL_CLEARING"/>
    </condition>
  </extension>

  <!-- ============================================================
       Extension: trasferimento al numero di fallback (premi 1)
       Usa il gateway SIP e tech_prefix configurati dal Voice Bridge
  ============================================================ -->
  <extension name="transfer_to_fallback">
    <condition field="destination_number" expression="^transfer_to_fallback$">

      <action application="log" data="INFO [OVERFLOW-TRANSFER] Chiamata ${uuid} - trasferimento a ${fallback_transfer_number} via ${sip_gateway} (prefix=${tech_prefix})"/>

      <action application="clear_digit_action" data="overflow"/>

      <action application="playback" data="/opt/sounds/overflow/transferring.wav"/>
      <anti-action application="playback" data="tone_stream://%(200,100,600);%(200,100,500)"/>

      <action application="set" data="hangup_after_bridge=true"/>
      <action application="set" data="call_timeout=30"/>
      <action application="set" data="absolute_codec_string=PCMU,PCMA,G722"/>

      <action application="bridge" data="sofia/gateway/${sip_gateway}/${tech_prefix}${fallback_transfer_number}"/>

      <!-- Se il bridge fallisce, arriva qui -->
      <action application="log" data="WARNING [OVERFLOW-TRANSFER] Bridge fallito per ${fallback_transfer_number}"/>
      <action application="playback" data="/opt/sounds/overflow/transfer_failed.wav"/>
      <anti-action application="playback" data="tone_stream://%(300,200,400)"/>
      <action application="hangup" data="NORMAL_CLEARING"/>

    </condition>
  </extension>

</include>
