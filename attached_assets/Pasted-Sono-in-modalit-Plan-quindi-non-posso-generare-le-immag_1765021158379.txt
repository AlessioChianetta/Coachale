MODULO 4: DATABASE (Nuove Tabelle SQL)
-- Venditori umani (parallelo a client_sales_agents)
CREATE TABLE human_sellers (
  id VARCHAR PRIMARY KEY DEFAULT gen_random_uuid(),
  client_id VARCHAR REFERENCES users(id),
  seller_name TEXT NOT NULL,
  display_name TEXT NOT NULL,
  is_active BOOLEAN DEFAULT true,
  voice_config JSONB, -- config per TTS se serve
  created_at TIMESTAMP DEFAULT now()
);
-- Meeting video programmati
CREATE TABLE video_meetings (
  id VARCHAR PRIMARY KEY DEFAULT gen_random_uuid(),
  seller_id VARCHAR REFERENCES human_sellers(id),
  meeting_token VARCHAR UNIQUE NOT NULL, -- per URL
  playbook_id VARCHAR REFERENCES playbooks(id),
  scheduled_at TIMESTAMP,
  status TEXT DEFAULT 'scheduled', -- scheduled/active/completed
  created_at TIMESTAMP DEFAULT now()
);
-- Partecipanti meeting
CREATE TABLE video_meeting_participants (
  id VARCHAR PRIMARY KEY DEFAULT gen_random_uuid(),
  meeting_id VARCHAR REFERENCES video_meetings(id),
  name TEXT NOT NULL,
  role TEXT NOT NULL, -- 'host', 'guest', 'shadow'
  joined_at TIMESTAMP,
  left_at TIMESTAMP
);
-- Trascrizioni con speaker diarization
CREATE TABLE video_meeting_transcripts (
  id VARCHAR PRIMARY KEY DEFAULT gen_random_uuid(),
  meeting_id VARCHAR REFERENCES video_meetings(id),
  speaker_id VARCHAR, -- participant id
  speaker_name TEXT,
  text TEXT NOT NULL,
  timestamp_ms INTEGER, -- milliseconds from start
  sentiment TEXT, -- positive/neutral/negative
  created_at TIMESTAMP DEFAULT now()
);
-- Analytics video meeting
CREATE TABLE video_meeting_analytics (
  id VARCHAR PRIMARY KEY DEFAULT gen_random_uuid(),
  meeting_id VARCHAR REFERENCES video_meetings(id),
  duration_seconds INTEGER,
  talk_ratio REAL, -- 0-1 host talk percentage
  script_adherence REAL, -- 0-100%
  avg_sentiment_score REAL, -- 0-100
  objections_count INTEGER,
  objections_handled INTEGER,
  ai_summary TEXT,
  action_items JSONB,
  created_at TIMESTAMP DEFAULT now()
);
-- Playbooks/Scripts
CREATE TABLE playbooks (
  id VARCHAR PRIMARY KEY DEFAULT gen_random_uuid(),
  client_id VARCHAR REFERENCES users(id),
  name TEXT NOT NULL,
  type TEXT, -- 'discovery', 'demo', 'closing'
  phases JSONB NOT NULL, -- struttura fasi con checkpoints
  battle_cards JSONB, -- cards per obiezioni
  created_at TIMESTAMP DEFAULT now()
);
