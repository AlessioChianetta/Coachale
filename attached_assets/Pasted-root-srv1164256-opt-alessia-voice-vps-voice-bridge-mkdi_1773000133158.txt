cat > /etc/systemd/system/alessia-recordings-mount.service << 'EOF'
[Unit]
Description=Mount FreeSWITCH recordings from Docker container
After=docker.service
Requires=docker.service
[Service]
Type=oneshot
RemainAfterExit=yes
ExecStartPre=/bin/mkdir -p /opt/alessia-voice/vps-voice-bridge/recordings
ExecStart=/bin/bash -c 'MERGE=$(docker inspect freeswitch --format "{{.GraphDriver.Data.MergedDir}}") && mount --bind ${MERGE}/tmp/alessia_recordings /opt/alessia-voice/vps-voice-bridge/recordings'
ExecStop=/bin/umount /opt/alessia-voice/vps-voice-bridge/recordings
[Install]
WantedBy=multi-user.target
EOF
