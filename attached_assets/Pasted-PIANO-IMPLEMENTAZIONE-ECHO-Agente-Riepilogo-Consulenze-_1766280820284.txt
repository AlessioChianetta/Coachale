PARTE 4: DATABASE SCHEMA
4.1 Nuove Tabelle/Campi
-- Nuovi campi in consultations
ALTER TABLE consultations ADD COLUMN summary_email_status 
  VARCHAR(20) DEFAULT 'missing';
  -- Valori: 'missing', 'draft', 'approved', 'sent'
ALTER TABLE consultations ADD COLUMN summary_email_approved_at 
  TIMESTAMP;
ALTER TABLE consultations ADD COLUMN summary_email_sent_at 
  TIMESTAMP;
-- Nuovi campi in consultation_tasks
ALTER TABLE consultation_tasks ADD COLUMN source 
  VARCHAR(20) DEFAULT 'manual';
  -- Valori: 'manual', 'echo_ai', 'followup'
ALTER TABLE consultation_tasks ADD COLUMN status 
  VARCHAR(20) DEFAULT 'active';
  -- Valori: 'draft', 'active', 'completed', 'cancelled'
ALTER TABLE consultation_tasks ADD COLUMN linked_email_id 
  VARCHAR(36);
  -- Riferimento all'email che ha generato questa task
