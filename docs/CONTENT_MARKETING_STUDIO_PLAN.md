```sql
    start_date DATE,
    end_date DATE,
```

```typescript
// Middleware per tutte le route Content Studio
const contentStudioAuth = async (req, res, next) => {
  // 1. Verifica JWT token
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'UNAUTHORIZED' });
  
  // 2. Decodifica e verifica
  const decoded = jwt.verify(token, process.env.JWT_SECRET);
  
  // 3. Verifica ruolo (solo consultant o admin)
  if (!['consultant', 'super_admin'].includes(decoded.role)) {
    return res.status(403).json({ error: 'FORBIDDEN' });
  }
  
  // 4. Attach user a request
  req.user = decoded;
  next();
};

// Middleware per verificare ownership risorse
const verifyOwnership = (resourceTable) => async (req, res, next) => {
  const resourceId = req.params.id;
  const resource = await db.query(
    `SELECT consultant_id FROM ${resourceTable} WHERE id = $1`,
    [resourceId]
  );
  
  if (!resource || resource.consultant_id !== req.user.id) {
    return res.status(404).json({ error: 'NOT_FOUND' });
  }
  
  next();
};
```
